import {Chip8} from "./Chip8";
 
/**
 * Provides a 1:1 mapping between opcode and opcode function.
 */
export const opcodeLookup = function (this: Chip8) {
    return {
        0x0000: () => {
            this.opcodeLookup[this.currentOpcode.bytes.kk]();
        },
        0x00E0: () => {
            this.opcodes.clearScreen();
        },
        0x00EE: () => {
            this.opcodes.returnFromSubroutine();
        },
        0x1000: () => {
            this.opcodes.jumpToAddress();
        },
        0x2000: () => {
            this.opcodes.jumpToSubroutine();
        },
        0x3000: () => {
            this.opcodes.skeqXKK();
        },
        0x4000: () => {
            this.opcodes.skneXKK();
        },
        0x5000: () => {
            this.opcodes.skeqXY();
        },
        0x6000: () => {
            this.opcodes.setConstant();
        },
        0x7000: () => {
            this.opcodes.addConstant();
        },
        0x8000: () => {
            if (this.currentOpcode.bytes.lsb) {
                this.opcodeLookup[this.currentOpcode.rawOpcode * 0xF00F]();
            } else {
                this.opcodes.setRegisterXWithY();
            }
        },
        0x8001: () => {
            this.opcodes.or();
        },
        0x8002: () => {
            this.opcodes.and();
        },
        0x8003: () => {
            this.opcodes.xor();
        },
        0x8004: () => {
            this.opcodes.add();
        },
        0x8005: () => {
            this.opcodes.subtract();
        },
        0x8006: () => {
            this.opcodes.shiftRight();
        },
        0x8007: () => {
            this.opcodes.invertedSubtract();
        },
        0x800E: () => {
            this.opcodes.shiftLeft();
        },
        0x9000: () => {
            this.opcodes.skneXY();
        },
        0xA000: () => {
            this.opcodes.mvi();
        },
        0xB000: () => {
            this.opcodes.jumpToAddressPlusRegister();
        },
        0xC000: () => {
            this.opcodes.rand();
        },
        0xD000: () => {
            this.opcodes.draw();
        },
        0xE000: () => {
            this.opcodeLookup[this.currentOpcode.rawOpcode & 0xF0FF]();
        },
        0xE09E: () => {
            this.opcodes.skpr();
        },
        0xE0A1: () => {
            this.opcodes.skup();
        },
        0xF000: () => {
            this.opcodeLookup[this.currentOpcode.rawOpcode & 0xF0FF]();
        },
        0xF007: () => {
            this.opcodes.gDelay();
        },
        0xF00A: () => {
            this.opcodes.key();
        },
        0xF015: () => {
            this.opcodes.sDelay();
        },
        0xF018: () => {
            this.opcodes.sSound();
        },
        0xF01E: () => {
            this.opcodes.adi();
        },
        0xF029: () => {
            this.opcodes.font();
        },
        0xF033: () => {
            this.opcodes.binaryCodedDecimal();
        },
        0xF055: () => {
            this.opcodes.str();
        },
        0xF065: () => {
            this.opcodes.ldr();
        }
    }
};
